desired_constructors_min = {
	modifier = {
		set = 1
		OR = {
			is_country_type = default
			is_country_type = awakened_fallen_empire
			# mod
			is_country_type = hfe_drone
			is_country_type = hfe_pirate
			is_country_type = hfe_fallen
		}
	}
	modifier = {
		factor = 2
		or = {
			is_country_type = default
			is_country_type = hfe_pirate
			is_country_type = hfe_drone
		}
		controlled_systems > 3
		years_passed < 100
	}
}
