hfe_effect_media_random = {
	if = {
		limit = {
			is_gestalt = yes
			not = {
				has_valid_civic = civic_machine_servitor
			}
		}
	}
	else = {
		random_list = {
			10 = {
				hfe_effect_media_negative = yes
			}
			10 = {
				hfe_effect_media_positive = yes
			}
		}
	}
}

hfe_effect_media_negative = {

		limit = {
			is_gestalt = yes
			not = {

			}
		}

	else_if = {
		limit = {
			has_modifier = media_coverage_negative_high
		}
		remove_modifier = media_coverage_negative_high
		add_modifier = {
			modifier = media_coverage_negative_high
			days = @hfe_media_days
		}

hfe_effect_authoritarian_attraction.0 = {
	if = {
		limit = {
			not = {
				has_modifier = hfe_authoritarian_attraction.5
			}
			is_gestalt = no
		}
		switch = {
			trigger = has_modifier
			hfe_authoritarian_attraction.4 = {
				remove_modifier = hfe_authoritarian_attraction.4
				add_modifier = {
					modifier = hfe_authoritarian_attraction.5
					years = 1
				}
			}
			hfe_authoritarian_attraction.3 = {
				remove_modifier = hfe_authoritarian_attraction.3
				add_modifier = {
					modifier = hfe_authoritarian_attraction.4
					years = 1
				}
			}
			hfe_authoritarian_attraction.2 = {
				remove_modifier = hfe_authoritarian_attraction.2
				add_modifier = {
					modifier = hfe_authoritarian_attraction.3
					years = 1
				}
			}
			hfe_authoritarian_attraction.1 = {
				remove_modifier = hfe_authoritarian_attraction.1
				add_modifier = {
					modifier = hfe_authoritarian_attraction.2
					years = 1
				}
			}
			default = {
				add_modifier = {
					modifier = hfe_authoritarian_attraction.0
					years = 1
				}
			}
		}
	}
}
